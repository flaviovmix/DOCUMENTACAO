DepartamentoPessoaSelect novo arquivo

0:00 / 0:00

Arquivo novo e o maior da sequência do LinkBox. É o modal que abre quando o usuário clica no botão do campo "Responsável" no form do Produto: uma lista de pessoas com ordenação, busca rápida e paginação. Ao clicar numa pessoa, o Select chama de volta uma URL informada pelo LinkBox (o callback) e o Form do Produto recebe o id_pessoa pra gravar no Bean.

Parece grande, mas quase tudo é um padrão repetido do Jasap pra "Select de entidade": ListView configurada pra chamar um callback ao clicar numa linha, filtro salvo em sessão pra lembrar a busca entre reordenações, e 3 actions (principal, Sort, QuickSearch) pra cobrir os três cenários de renderização.

import br.jasap.core.Effect;
import br.jasap.effect.Response;
import br.jasap.gui.Bar;
import br.jasap.gui.JasapPage;
import br.jasap.gui.ListColumn;
import br.jasap.gui.ListLine;
import br.jasap.gui.ListView;

import br.jasap.gui.form.LinkBox;

import br.xt.acore.view.XtPage;

public class DepartamentoPessoaSelect extends DepartamentoPessoaAction {

    @Override
    public Effect execute() throws Exception {
        getSession().remove(S_ROOT);
        render();
        return new Response();
    }

    public static class Sort extends DepartamentoPessoaSelect {
        @Override
        public Effect execute() throws Exception {
            update(listV().getDIV_HEADER(),   listV().getHeader());
            update(listV().getDIV_BODY(),     listV().getBody());
            update(listV().getDIV_NAVIGATE(), listV().getNavForm());
            return new Response();
        }
    }

    public static class QuickSearch extends DepartamentoPessoaSelect {
        @Override
        public Effect execute() throws Exception {
            getFiltroS().setQs_pessoa(getInput().getString(DepartamentoPessoaBean.QS_PESSOA));
            update(listV().getDIV_BODY(),     listV().getBody());
            update(listV().getDIV_NAVIGATE(), listV().getNavForm());
            return new Response();
        }
    }

    public void render() throws Exception {
        XtPage page = new XtPage(getManager());
        if (isAjaxCall()) {
            update(JasapPage.DIV_WINDOW, page.content(window().toHtml()));
            eval(Js.CLOSE_SUB_WINDOWS);
        } else {
            page.getTable().setBorder(4).rowC("100%")
                    .setContent(page.content(window().toHtml()))
                    .setStyle(ui().modalBorder());
            page.setWinTitle("Selecionar Pessoa");
            getOutput().write(this, page);
        }
    }

    public Table window() throws Exception {
        Table w = new Table(getManager()).setSize("100%", "100%");
        w.rowC("1%",  JasapPage.DIV_TITLE, ui().title("SELEÇÃO DE PESSOA"));
        w.rowC("99%").setId(JasapPage.DIV_MASTER).setContent(listV()).table();
        w.rowC("1%",  null, ui().line());
        w.rowC("1%",  null, qs_pessoa());
        w.rowC("1%",  JasapPage.DIV_BOTTOM, br());
        return w;
    }

    public String qs_pessoa() throws Exception {
        Text qs = ui().text(DepartamentoPessoaBean.QS_PESSOA)
                .setLabel("Consulta")
                .setStyle("width:300")
                .setMaxlength(300)
                .setValue(getFiltroS().getQs_pessoa())
                .setOnkeyup(Js.pressEnter(
                        link(QuickSearch.class)
                                .putScript(DepartamentoPessoaBean.QS_PESSOA, Js.SELF_VALUE)
                                .putInteger(listV().getPAGE(), 0)
                                .ajax()));

        Table aux = new Table(getManager(), "100%", "30")
                .setStyle(ui().bgDark())
                .rowC()
                .setAlign(Table.ALIGN_CENTER)
                .setVerticalAlign(Table.ALIGN_MIDDLE)
                .setContent(qs)
                .table();

        return aux.toHtml();
    }

    private Bar br() throws Exception {
        return ui().bar().addCenter(listV().nav(listV().getNavForm()));
    }

    private ListView lv = null;
    public ListView listV() throws Exception {
        if (lv == null) {
            lv = ui().lView();
            lv.setCallBackURL(getSession().getString(LinkBox.CALLBACK_URL.concat("_PESSOA"), getInput()))
                    .setSortAct(url(Sort.class))
                    .setPageAction(url(Sort.class))
                    .setPage(getSession().getInteger(S_LIST.concat(lv.getPAGE()), getInput()))
                    .setOrderBy(getSession().getString(S_LIST.concat(lv.getORDER_BY()), getInput()))
                    .setSort(getSession().getString(S_LIST.concat(lv.getSORT()), getInput()))
                    .ajax();

            if (lv.getData().getOrderBy() == null) {
                lv.getData().setOrderBy(DepartamentoPessoaBean.NOME_PESSOA);
                lv.getData().setSort(JasapList.SORT_ASC);
            }

            lv.setFiltro(getFiltroS());
            getFactory().departamento().pesModel().daoList(lv.getData());

            ListColumn col_nome    = lv.newColumn("Nome").setOrderBy(DepartamentoPessoaBean.NOME_PESSOA).setPadding(";padding:10 8 10 8;");
            ListColumn col_apelido = lv.newColumn("Apelido").setOrderBy(DepartamentoPessoaBean.APELIDO_PESSOA).setWidth(180).setPadding(";padding:10 8 10 8;");

            while (lv.hasNext()) {
                DepartamentoPessoaBean lb = (DepartamentoPessoaBean) lv.next();
                ListLine line = lv.createLine()
                        .setOnclick(link(lv.getCallBackURL())
                                .putInteger(DepartamentoPessoaBean.ID_PESSOA, lb.getId_pessoa())
                                .noWait().ajax());
                col_nome.setContent(lb.getNome_pessoa());
                col_apelido.setContent(lb.getApelido_pessoa());
                lv.addLine(line);
            }
        }
        return lv;
    }

    public DepartamentoPessoaWBean getFiltroS() throws Exception {
        DepartamentoPessoaWBean filtro = (DepartamentoPessoaWBean) getSession().getObject(S_FILTRO);
        if (filtro == null) {
            filtro = pesWBean();
            getSession().addObj(S_FILTRO, filtro);
        }
        return filtro;
    }

    public static final String S_ROOT   = ROOT.concat("__PESSOA_SELECT/");
    public static final String S_LIST   = S_ROOT.concat("__LIST/");
    public static final String S_FILTRO = S_LIST.concat("__FILTRO");

}
Visão Geral — a anatomia de um Select

Um Select no Jasap é uma ListView em miniatura dentro de um modal. Parecido com uma List comum (tem ordenação, paginação e busca), mas com duas diferenças importantes:

  • Ao clicar numa linha, em vez de abrir o Form daquela linha, chama uma URL de callback informada por quem abriu o Select (no nosso caso, o LinkBox do Form do Produto)
  • A ordenação/busca/página ficam salvas em sessão, num namespace próprio (S_ROOT) — pra não conflitar com a List principal da entidade nem com outros Selects

O arquivo tem 3 entry points (actions acessíveis por URL):

ActionQuando é chamadaO que faz
DepartamentoPessoaSelect (principal)Quando o LinkBox abre o modal pela primeira vezLimpa a sessão do Select (zera busca/página anteriores) e renderiza o modal cheio
.SortAo clicar no cabeçalho duma coluna ou trocar de páginaRe-renderiza só a ListView (header, body, navegação) via Ajax — não redesenha o modal inteiro
.QuickSearchAo pressionar Enter no campo de buscaGrava o termo no filtro em sessão e re-renderiza body + navegação

O resto do arquivo são os métodos auxiliares que montam a tela: render(), window(), qs_pessoa(), br(), listV() e getFiltroS(). Plus 3 constantes de namespace de sessão.

Conceito — Como o Select conversa com o LinkBox (callback)

O LinkBox no Form do Produto, ao ser renderizado, grava na sessão a URL que deve ser chamada quando o usuário selecionar uma pessoa. A chave de sessão é LinkBox.CALLBACK_URL + "_PESSOA" (o sufixo "_PESSOA" vem do setCallBackID("_PESSOA") no LinkBox).

Quando o Select monta a ListView, ele lê essa URL da sessão:

lv.setCallBackURL(getSession().getString(LinkBox.CALLBACK_URL.concat("_PESSOA"), getInput()))

E cada linha da ListView tem um onclick que dispara essa URL passando o id_pessoa:

.setOnclick(link(lv.getCallBackURL())
        .putInteger(DepartamentoPessoaBean.ID_PESSOA, lb.getId_pessoa())
        .noWait().ajax());

Do lado do Produto, a URL apontada é a inner class Fk_pessoa_produto — ela recebe o id_pessoa, grava no bean e atualiza o texto do campo no form. Veja a próxima página da sequência.

ElementoPapel no fluxo

LinkBox.CALLBACK_URL.concat("_PESSOA") no SelectChave de sessão onde o LinkBox gravou a URL. O Select lê daqui

Conceito — Filtro cacheado em sessão (getFiltroS) 
public DepartamentoPessoaWBean getFiltroS() throws Exception {
    DepartamentoPessoaWBean filtro = (DepartamentoPessoaWBean) getSession().getObject(S_FILTRO);
    if (filtro == null) {
        filtro = pesWBean();
        getSession().addObj(S_FILTRO, filtro);
    }
    return filtro;
}

O filtro (o WBean) fica salvo em sessão sob a chave S_FILTRO. O padrão é:

  1. Primeira chamada ao Select → filtro não existe → cria um novo (pesWBean()) e salva
  2. Chamadas seguintes (Sort, QuickSearch, paginação) → retorna o filtro já salvo

Isso permite que a busca digitada no QuickSearch persista quando o usuário reordena uma coluna ou muda de página. Sem o cache, cada clique recriaria um filtro em branco e a busca sumiria.

A chave S_FILTRO é reinicializada na primeira ação (a principal DepartamentoPessoaSelect.execute() chama getSession().remove(S_ROOT) — como S_FILTRO começa com S_ROOT, ele é apagado junto). Assim, cada vez que o LinkBox abre o Select do zero, a busca anterior não vaza.

Inner classes Sort e QuickSearch — re-renderização parcial

As duas inner classes existem pra atualizar só pedaços da ListView sem redesenhar o modal inteiro. Isso deixa a interação fluida — o cabeçalho do modal, o campo de busca e os botões não precisam ser re-enviados só porque o usuário clicou numa coluna pra reordenar.

public static class Sort extends DepartamentoPessoaSelect {

        update(listV().getDIV_BODY(),     listV().getBody());

A Sort recarrega 3 pedaços: cabeçalho das colunas (setinhas de ordenação), corpo (linhas) e navegação (botões de página). O Sort também é usado como ação de paginação — por isso atualiza a navegação.

public static class QuickSearch extends DepartamentoPessoaSelect {
    @Override
    public Effect execute() throws Exception {
        getFiltroS().setQs_pessoa(getInput().getString(DepartamentoPessoaBean.QS_PESSOA));
        update(listV().getDIV_BODY(),     listV().getBody());
        update(listV().getDIV_NAVIGATE(), listV().getNavForm());
        return new Response();
    }
}

A QuickSearch grava o termo digitado no filtro em sessão (getFiltroS().setQs_pessoa(...)) e recarrega só corpo + navegação. O cabeçalho das colunas não muda — a ordem continua a mesma.

extends DepartamentoPessoaSelect permite que essas inner classes usem os mesmos métodos (listV(), getFiltroS()) e constantes (S_LIST, S_FILTRO) da classe pai.

O padrão memoization (if (lv == null)) garante que a ListView é montada uma única vez por requisição. Sort, QuickSearch e render() podem chamar listV() várias vezes e recebem sempre o mesmo objeto já configurado.
