Formulário de cadastro e edição de Produto. A mesma tela serve para inserção e edição — o modo atual determina quais botões são visíveis. Concentra seis inner classes: ShowInsert, ShowUpdate, Cancelar, Insert, Update e Delete.

package br.xt.app.departamento.produto;

import br.jasap.core.Effect;
import br.jasap.effect.Response;
import br.jasap.gui.Bar;
import br.jasap.gui.Button;
import br.jasap.gui.JasapPage;
import br.jasap.gui.Table;
import br.jasap.gui.Toast;
import br.jasap.gui.form.Check;
import br.jasap.gui.form.Form;
import br.jasap.gui.form.Text;
import br.jasap.gui.form.Textarea;
import br.jasap.util.DomainValue;
import br.jasap.util.JasapFunctions;
import br.jasap.util.JasapList;
import br.jasap.util.Js;
import br.jasap.util.exceptions.SQLConstraintException;
import br.jasap.util.filters.TransactionFilter;
import br.xt.acore.view.XtPage;

public class DepartamentoProdutoForm extends DepartamentoProdutoAction {

    @Override
    public Effect execute() throws Exception {
        render();
        return new Response();
    }

    public static class ShowInsert extends DepartamentoProdutoForm {
        @Override
        public Effect execute() throws Exception {
            setInsert(FORM);
            getSession().remove(ROOT.concat(DepartamentoProdutoBean.ID_PRO));
            proBean().setInsert_chk(getInput().getInteger(DepartamentoProdutoBean.INSERT_CHK));
            render();
            return new Response();
        }
    }

    public static class ShowUpdate extends DepartamentoProdutoForm {
        @Override
        public Effect execute() throws Exception {
            setUpdate(FORM);
            getSession().addInt(ROOT.concat(DepartamentoProdutoBean.ID_PRO), proBean().getId_pro());
            getFactory().departamento().proModel().daoSingle(proBean());
            render();
            return new Response();
        }
    }

    public static class Cancelar extends DepartamentoProdutoForm {
        @Override
        public Effect execute() throws Exception {
            evalParent(Js.CLOSE_SUB_WINDOWS);
            return new Response();
        }
    }

    public static class Insert extends DepartamentoProdutoAction {
        public Insert() { super.getFilters().add(new TransactionFilter()); }
        @Override
        public Effect execute() throws Exception {
            Integer chk = proBean().getInsert_chk();
            proBean().setInsert_chk(null);
            getFactory().departamento().proModel().insert(proBean());
            if (JasapFunctions.equals(chk, 1)) {
                eval(link(ShowInsert.class)
                        .putInteger(DepartamentoProdutoBean.INSERT_CHK, 1)
                        .noWait().ajax());
                evalParent(link(DepartamentoProdutoList.Sort.class).noWait().ajax(false));
            } else {
                evalParent(Js.CLOSE_SUB_WINDOWS);
            }
            return new Response();
        }
    }

    public static class Update extends DepartamentoProdutoAction {
        @Override
        public Effect execute() throws Exception {
            getFactory().departamento().proModel().update(proBean());
            evalParent(Js.CLOSE_SUB_WINDOWS);
            return new Response();
        }
    }

    public static class Delete extends DepartamentoProdutoAction {
        @Override
        public Effect execute() throws Exception {
            try {
                if (JasapFunctions.equals(getInput().getInteger(CONFIRM), 1)) {
                    getFactory().departamento().proModel().daoDelete(proBean());
                    getSession().remove(ROOT.concat(DepartamentoProdutoBean.ID_PRO));
                    evalParent(Js.CLOSE_SUB_WINDOWS);
                } else {
                    eval(Js.swalConfirm(
                            "Confirma a exclusão desse registro?",
                            submit(Delete.class).putInteger(CONFIRM, 1).ajax().toHtml(),
                            ""));
                }
                return new Response();
            } catch (SQLConstraintException e) {
                eval(new Toast("Falha ao excluir: esse registro está sendo usado em outra tabela.").fail());
                return new Response();
            }
        }
    }

    public void render() throws Exception {
        XtPage page = new XtPage(getManager());
        if (isAjaxCall()) {
            update(JasapPage.DIV_WINDOW, page.content(window().toHtml()));
            eval(Js.CLOSE_SUB_WINDOWS);
        } else {
            page.getTable().setBorder(4).rowC("100%")
                    .setContent(page.content(window().toHtml()))
                    .setStyle(ui().modalBorder());
            page.setWinTitle("Produto");
            getOutput().write(this, page);
        }
    }

    public Table window() throws Exception {
        Table w = new Table(getManager()).setSize("100%", "100%");
        w.rowC("1%",  JasapPage.DIV_TITLE, ui().title("CADASTRO DE PRODUTO"));
        w.rowC("1%",  null, ui().line());
        w.rowC("auto").setId(JasapPage.DIV_MASTER).setContent(form()).table();
        w.rowC("1%",  null, ui().line());
        w.rowC("1%",  JasapPage.DIV_BOTTOM, br());
        return w;
    }

    protected Bar br() throws Exception {
        Button salvar = ui().button("&nbsp;&nbsp;Salvar&nbsp;&nbsp;")
                .setCss("btn btn-success btn-lg").setNoSize()
                .setOnClick(submit(Insert.class).validate().toHtml())
                .setVisible(isInsert(FORM));

        Button salvarUpdate = ui().button("&nbsp;&nbsp;Salvar&nbsp;&nbsp;")
                .setCss("btn btn-success btn-lg").setNoSize()
                .setOnClick(submit(Update.class).validate().toHtml())
                .setVisible(isUpdate(FORM));

        Button excluir = ui().button("&nbsp;&nbsp;Excluir&nbsp;&nbsp;")
                .setCss("btn btn-danger btn-lg").setNoSize()
                .setOnClick(submit(Delete.class).toHtml())
                .setVisible(isUpdate(FORM));

        Button cancelar = ui().button("&nbsp;&nbsp;Cancelar&nbsp;&nbsp;")
                .setCss("btn btn-primary btn-lg").setNoSize()
                .setOnClick(link(Cancelar.class).ajax());

        Bar bar = ui().bar();
        bar.addRight(cancelar);
        bar.addRight("&nbsp;");
        bar.addRight(salvar);
        bar.addRight(excluir);
        bar.addRight("&nbsp;");
        bar.addRight(salvarUpdate);
        return bar;
    }

    public String form() throws Exception {
        Form frm = ui().form();
        frm.addHidden(DepartamentoProdutoBean.ID_PRO, proBean().getId_pro());
        frm.line().add(nome_pro(),   "150");
        frm.line().add(vl_pro(),     "150");
        frm.line().add(qtd_pro(),    "150");
        frm.line().add(obs_pro(),    "150");
        frm.line().add(insert_chk(), "150");
        return frm.getTable().toHtml();
    }

    protected Text nome_pro = null;
    public Text nome_pro() throws Exception {
        if (nome_pro == null) {
            nome_pro = ui().text(DepartamentoProdutoBean.NOME_PRO)
                    .setLabel("Nome").setStyle("width:350;height:25")
                    .setMaxlength(100).setRequired(true)
                    .setValue(proBean().getNome_pro());
        }
        return nome_pro;
    }

    protected Text vl_pro = null;
    public Text vl_pro() throws Exception {
        if (vl_pro == null) {
            vl_pro = ui().text(DepartamentoProdutoBean.VL_PRO)
                    .setLabel("Valor").setStyle("width:120;text-align:right;height:25")
                    .setValue(proBean().getVl_pro());
        }
        return vl_pro;
    }

    protected Text qtd_pro = null;
    public Text qtd_pro() throws Exception {
        if (qtd_pro == null) {
            qtd_pro = ui().text(DepartamentoProdutoBean.QTD_PRO)
                    .setLabel("Quantidade").setStyle("width:80;text-align:right;height:25")
                    .setValue(proBean().getQtd_pro());
        }
        return qtd_pro;
    }

    protected Textarea obs_pro = null;
    public Textarea obs_pro() throws Exception {
        if (obs_pro == null) {
            obs_pro = ui().textArea(DepartamentoProdutoBean.OBS_PRO, "width:530;height:80")
                    .setLabel("Observações").setValue(proBean().getObs_pro());
        }
        return obs_pro;
    }

    protected Check insert_chk = null;
    public Check insert_chk() throws Exception {
        if (insert_chk == null) {
            JasapList aux = new JasapList();
            aux.getList().add(new DomainValue(1, "Continuar inserindo"));
            aux.setSelected(proBean().getInsert_chk());
            insert_chk = new Check(getManager(), DepartamentoProdutoBean.INSERT_CHK)
                    .setLabel("&nbsp;")
                    .setLoad(isInsert(FORM))
                    .setValue(aux);
        }
        return insert_chk;
    }

    public static String FORM    = ROOT.concat("__FORM");
    public static String CONFIRM = ROOT.concat("__CONFIRM");

}

ShowInsert e ShowUpdate — preparar o formulário

O formulário nunca é aberto por execute() diretamente — sempre por ShowInsert (modo inserção) ou ShowUpdate (modo edição). A diferença entre os dois está no que fazem antes de chamar render():

ActionO que faz
ShowInsertsetInsert(FORM) — define modo inserção na sessão. Remove o ID da sessão para garantir bean limpo. Lê insert_chk do input para preservar estado do checkbox ao reabrir
ShowUpdatesetUpdate(FORM) — define modo edição na sessão. Grava o ID do produto na sessão (id_pro() passa a funcionar). Chama daoSingle(proBean()) para carregar todos os campos do banco

A chave FORM guarda o modo atual. isInsert(FORM) e isUpdate(FORM) consultam essa sessão para decidir quais botões mostrar em br().

Insert, Update e Delete — por que estendem DepartamentoProdutoAction?

Por convenção, as actions que escrevem no banco estendem diretamente DepartamentoProdutoAction (não o Form). Isso evita que render() seja herdado e chamado acidentalmente. A hierarquia é: DepartamentoProdutoAction → Insert, Update, Delete.

Insert — TransactionFilter e insert_chk

TransactionFilter é adicionado no construtor. Ele abre uma transação antes de execute() e faz commit ao final; se lançar exceção, faz rollback automaticamente.

O campo insert_chk precisa de cuidado especial:

LinhaPor quê
Integer chk = proBean().getInsert_chk()Lê o estado do checkbox ANTES de limpar — depois do insert precisamos saber se o usuário quer continuar ou fechar
proBean().setInsert_chk(null)OBRIGATÓRIO antes do insert. insert_chk é transiente (sem @DBInfo), mas se não for nulo o framework pode tentar incluí-lo no SQL e gerar erro de coluna inexistente
if (JasapFunctions.equals(chk, 1))Se marcado: reabre o formulário vazio no modal E atualiza a lista no pai via evalParent. Se não: fecha o modal

form() — addHidden e campos

frm.addHidden(ID_PRO, proBean().getId_pro()) inclui o ID como campo oculto. É assim que Insert e Update recebem o ID — submit() envia todos os campos do formulário, incluindo os hidden.

Na inserção o valor é null (bean limpo); na edição é o ID carregado pelo ShowUpdate.

Campos — padrão lazy e setValue

Todos os campos usam o mesmo padrão lazy do proBean() na Action (private campo = null / if (campo == null)). O setValue(proBean().getCampo()) preenche com o valor atual:

  • Modo inserção → bean vazio → campo aparece vazio
  • Modo edição → bean carregado pelo daoSingle → campo aparece com o valor do banco

insert_chk — Continuar inserindo

O checkbox "Continuar inserindo" aparece apenas no modo inserção (setLoad(isInsert(FORM))). Quando marcado e o usuário clica "Salvar", o Insert:

  1. Salva o registro normalmente
  2. Reabre o formulário vazio no mesmo modal — sem fechar a janela
  3. Atualiza a lista no frame pai — o novo registro já aparece na lista

aux.setSelected(proBean().getInsert_chk()) preserva o estado marcado ao reabrir o formulário, para que o usuário não precise marcar o checkbox novamente a cada inserção.
